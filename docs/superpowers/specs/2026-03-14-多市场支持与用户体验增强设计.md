本设计扩展AI Hedge Fund系统以支持多个市场（A股、港股、期货/商品），并通过投资组合管理和实时监控功能增强用户体验。核心原则是**保持现有18个代理的LangGraph架构不变**，将数据源适配器和UI增强作为独立模块添加。

**核心目标：**
1. 使现有代理能够分析A股、港股和商品，无需修改代码
2. 添加投资组合追踪、交易历史和卖出提醒功能
3. 增强Web Dashboard，支持实时监控
4. 保持与现有功能的向后兼容性

```python
# src/markets/base.py
from abc import ABC, abstractmethod
from typing import List, Optional
from datetime import datetime

class MarketAdapter(ABC):
    """所有市场数据适配器的基类"""

    @abstractmethod
    def supports_ticker(self, ticker: str) -> bool:
        """检查此适配器是否支持给定的ticker格式"""
        pass

    @abstractmethod
    def get_prices(
        self,
        ticker: str,
        start_date: str,
        end_date: str
    ) -> List[dict]:
        """
        获取历史价格数据

        返回:
            包含以下键的字典列表: date, open, high, low, close, volume
        """
        pass

    @abstractmethod
    def get_company_news(
        self,
        ticker: str,
        end_date: str,
        limit: int
    ) -> List[dict]:
        """
        获取公司新闻

        返回:
            包含以下键的字典列表: title, published, source, sentiment
        """
        pass

    @abstractmethod
    def get_financial_metrics(
        self,
        ticker: str,
        end_date: str
    ) -> dict:
        """
        获取财务指标

        返回:
            包含以下键的字典: pe_ratio, market_cap, revenue等
        """
        pass
```

### 1.4 市场路由器实现

```python
# src/markets/router.py
from typing import List
from .base import MarketAdapter
from .us_stock import USStockAdapter
from .cn_stock import CNStockAdapter
from .hk_stock import HKStockAdapter
from .commodity import CommodityAdapter

class MarketRouter:
    """将ticker路由到对应的市场适配器"""

    def __init__(self):
        # 顺序很重要：先检查特定模式，最后检查默认
        self.adapters: List[MarketAdapter] = [
            CNStockAdapter(),    # 先检查A股
            HKStockAdapter(),    # 然后港股
            CommodityAdapter(),  # 然后商品
            USStockAdapter(),    # 美股作为默认
        ]

    def route(self, ticker: str) -> MarketAdapter:
        """
        找到给定ticker的适配器

        参数:
            ticker: 股票代码

        返回:
            MarketAdapter实例

        异常:
            ValueError: 如果没有适配器支持该ticker
        """
        for adapter in self.adapters:
            if adapter.supports_ticker(ticker):
                return adapter

        raise ValueError(f"未找到支持该ticker的适配器: {ticker}")

    def get_prices(self, ticker: str, start_date: str, end_date: str):
        """便捷方法：路由并获取价格"""
        adapter = self.route(ticker)
        return adapter.get_prices(ticker, start_date, end_date)

    def get_company_news(self, ticker: str, end_date: str, limit: int):
        """便捷方法：路由并获取新闻"""
        adapter = self.route(ticker)
        return adapter.get_company_news(ticker, end_date, limit)

    def get_financial_metrics(self, ticker: str, end_date: str):
        """便捷方法：路由并获取财务指标"""
        adapter = self.route(ticker)
        return adapter.get_financial_metrics(ticker, end_date)
```

```python
# src/markets/cn_stock.py
import akshare as ak
from datetime import datetime
from typing import List
from .base import MarketAdapter

class CNStockAdapter(MarketAdapter):
    """A股市场适配器，使用AkShare"""

    def supports_ticker(self, ticker: str) -> bool:
        """支持格式: 600000.SH 或 000001.SZ"""
        return ticker.endswith('.SH') or ticker.endswith('.SZ')

    def get_prices(self, ticker: str, start_date: str, end_date: str) -> List[dict]:
        """
        使用AkShare获取A股历史价格

        数据源: akshare.stock_zh_a_hist()
        """
        symbol = ticker.split('.')[0]  # 提取代码（如 "600000"）

        # AkShare需要日期格式: YYYYMMDD
        start_str = start_date.replace('-', '')
        end_str = end_date.replace('-', '')

        df = ak.stock_zh_a_hist(
            symbol=symbol,
            start_date=start_str,
            end_date=end_str,
            adjust="qfq"  # 前复权（考虑拆股/分红）
        )

        # 转换为标准格式
        return self._convert_prices_to_standard(df)

    def get_company_news(self, ticker: str, end_date: str, limit: int) -> List[dict]:
        """
        从多个来源获取A股新闻

        来源:
        1. 东方财富新闻 (ak.stock_news_em)
        2. Google News RSS（备用）
        """
        symbol = ticker.split('.')[0]
        news_list = []

        try:
            # 主要来源：东方财富新闻
            df = ak.stock_news_em(symbol=symbol)
            news_list = self._convert_news_to_standard(df, limit)
        except Exception as e:
            # 备用：Google News RSS
            from src.tools.news_sources import GoogleNewsRSS
            rss = GoogleNewsRSS()
            news_list = rss.fetch_news(ticker, limit)

        return news_list

    def get_financial_metrics(self, ticker: str, end_date: str) -> dict:
        """
        获取A股财务指标

        数据源: akshare.stock_financial_analysis_indicator()
        """
        symbol = ticker.split('.')[0]
        df = ak.stock_financial_analysis_indicator(symbol=symbol)

        # 获取最新指标
        latest = df.iloc[-1]

        return {
            'pe_ratio': float(latest.get('pe_ratio', 0)),
            'pb_ratio': float(latest.get('pb_ratio', 0)),
            'market_cap': float(latest.get('market_cap', 0)),
            'revenue': float(latest.get('revenue', 0)),
            'net_profit': float(latest.get('net_profit', 0)),
        }

    def _convert_prices_to_standard(self, df) -> List[dict]:
        """将AkShare DataFrame转换为标准格式"""
        result = []
        for _, row in df.iterrows():
            result.append({
                'date': row['日期'],
                'open': float(row['开盘']),
                'high': float(row['最高']),
                'low': float(row['最低']),
                'close': float(row['收盘']),
                'volume': int(row['成交量']),
            })
        return result

    def _convert_news_to_standard(self, df, limit: int) -> List[dict]:
        """将AkShare新闻DataFrame转换为标准格式"""
        result = []
        for _, row in df.head(limit).iterrows():
            result.append({
                'title': row['新闻标题'],
                'published': row['发布时间'],
                'source': row['新闻来源'],
                'sentiment': None,  # 由LLM分析
            })
        return result
```

```python
import yfinance as yf
from typing import List
from .base import MarketAdapter

class CommodityAdapter(MarketAdapter):
    """商品和期货适配器"""

    SUPPORTED_COMMODITIES = {
        'GC=F': '黄金期货',
        'SI=F': '白银期货',
        'CL=F': '原油期货',
        'XAUUSD': '黄金现货',
        'XAGUSD': '白银现货',
    }

        return '=F' in ticker or ticker in ['XAUUSD', 'XAGUSD']

    def get_prices(self, ticker: str, start_date: str, end_date: str) -> List[dict]:
        """
        获取商品价格

        来源: Yahoo Finance（支持大多数期货）
        """
        df = yf.download(ticker, start=start_date, end=end_date, progress=False)
        return self._convert_to_standard(df)

    def get_company_news(self, ticker: str, end_date: str, limit: int) -> List[dict]:
        """
        获取商品相关新闻

        来源: Google News RSS，使用商品特定关键词
        """
        from src.tools.news_sources import GoogleNewsRSS
        rss = GoogleNewsRSS()

        # 将ticker映射到搜索关键词
        keywords = {
            'GC=F': 'gold price',
            'SI=F': 'silver price',
            'CL=F': 'crude oil price',
            'XAUUSD': 'gold',
            'XAGUSD': 'silver',
        }

        search_term = keywords.get(ticker, ticker)
        return rss.fetch_news(search_term, limit)

    def get_financial_metrics(self, ticker: str, end_date: str) -> dict:
        """
        商品没有财务指标
        返回空字典
        """
        return {}

    def _convert_to_standard(self, df) -> List[dict]:
        """转换为标准格式"""
        result = []
        for date, row in df.iterrows():
            result.append({
                'date': date.strftime('%Y-%m-%d'),

                'high': float(row['High']),
```

```python
# src/tools/news_sources.py (新文件)
import feedparser
from typing import List
from datetime import datetime

class GoogleNewsRSS:
    """
    Google News RSS订阅源获取器

    优势:
    - 免费，无需API密钥
    - 无限请求
    - 支持所有语言
    - 实时新闻
    """

    BASE_URL = "https://news.google.com/rss/search"

    def fetch_news(self, query: str, limit: int = 20, language: str = 'zh-CN') -> List[dict]:
        """
        从Google News RSS获取新闻

        参数:
            query: 搜索查询（ticker或关键词）
            limit: 最大文章数
            language: 语言代码（zh-CN中文, en英文）

        返回:
            新闻字典列表
        """
        # 构建RSS URL
        search_query = f"{query} 股票" if language == 'zh-CN' else f"{query} stock"
        params = {
            'q': search_query,
            'hl': language,
            'gl': 'CN' if language == 'zh-CN' else 'US',
            'ceid': 'CN:zh-Hans' if language == 'zh-CN' else 'US:en'
        }

        url = f"{self.BASE_URL}?{'&'.join(f'{k}={v}' for k, v in params.items())}"

        # 解析RSS订阅
        feed = feedparser.parse(url)

        # 转换为标准格式
        news_list = []
        for entry in feed.entries[:limit]:
            news_list.append({
                'title': entry.title,
                'published': self._parse_date(entry.published),
                'source': self._extract_source(entry),
                'link': entry.link,
                'sentiment': None,  # 由LLM分析
            })

        return news_list

    def _parse_date(self, date_str: str) -> str:
        """将RSS日期解析为ISO格式"""
        try:
            dt = datetime.strptime(date_str, '%a, %d %b %Y %H:%M:%S %Z')
            return dt.isoformat()
        except:
            return date_str

    def _extract_source(self, entry) -> str:
        """从条目中提取来源名称"""
        # Google News在标题中包含来源，如 "来源 - 文章标题"
        title = entry.title
        if ' - ' in title:
            return title.split(' - ')[0]
        return 'Google News'
```
