```python
"""
市场数据适配器基类

定义所有市场适配器必须实现的统一接口
"""
from abc import ABC, abstractmethod
from typing import List, Dict, Optional


class MarketAdapter(ABC):
    """
    市场数据适配器抽象基类

    所有市场适配器（美股、A股、港股、商品）都必须继承此类
    并实现所有抽象方法，以提供统一的数据访问接口
    """

    @abstractmethod
    def supports_ticker(self, ticker: str) -> bool:
        """
        检查此适配器是否支持给定的ticker格式

        Args:
            ticker: 股票/商品代码，如 "AAPL", "600000.SH", "0700.HK", "GC=F"

        Returns:
            bool: True表示支持，False表示不支持

        Examples:
            >>> adapter.supports_ticker("600000.SH")
            True
            >>> adapter.supports_ticker("AAPL")
            False
        """
        pass

    @abstractmethod
    def get_prices(
        self,
        ticker: str,
        start_date: str,
        end_date: str
    ) -> List[Dict]:
        """
        获取历史价格数据

        Args:
            ticker: 股票/商品代码
            start_date: 开始日期，格式 "YYYY-MM-DD"
            end_date: 结束日期，格式 "YYYY-MM-DD"

        Returns:
            List[Dict]: 价格数据列表，每个字典包含:
                - date: str, 日期 "YYYY-MM-DD"
                - open: float, 开盘价
                - high: float, 最高价
                - low: float, 最低价
                - close: float, 收盘价
                - volume: int, 成交量

        Raises:
            ValueError: ticker格式不支持
            Exception: 数据获取失败
        """
        pass

    @abstractmethod
    def get_company_news(
        self,
        ticker: str,
        end_date: str,
        limit: int
    ) -> List[Dict]:
        """
        获取公司/商品相关新闻

        Args:
            ticker: 股票/商品代码
            end_date: 截止日期，格式 "YYYY-MM-DD"
            limit: 最大新闻数量

        Returns:
            List[Dict]: 新闻列表，每个字典包含:
                - title: str, 新闻标题
                - published: str, 发布时间 ISO格式
                - source: str, 新闻来源
                - link: str, 新闻链接（可选）
                - sentiment: Optional[str], 情绪 (None表示未分析)

        Raises:
            ValueError: ticker格式不支持
            Exception: 数据获取失败
        """
        pass

    @abstractmethod
    def get_financial_metrics(
        self,
        ticker: str,
        end_date: str
    ) -> Dict:
        """
        获取财务指标

        Args:
            ticker: 股票代码
            end_date: 截止日期，格式 "YYYY-MM-DD"

        Returns:
            Dict: 财务指标字典，可能包含:
                - pe_ratio: float, 市盈率
                - pb_ratio: float, 市净率
                - market_cap: float, 市值
                - revenue: float, 营收
                - net_profit: float, 净利润
                注意：商品期货返回空字典{}

        Raises:
            ValueError: ticker格式不支持
            Exception: 数据获取失败
        """
        pass
```

```python
"""
市场路由器

根据ticker格式自动路由到对应的市场适配器
"""
from typing import List, Dict
from src.markets.base import MarketAdapter


class MarketRouter:
    """
    市场路由器

    根据ticker的格式特征，自动选择合适的市场适配器
    支持美股、A股、港股、商品期货等多个市场
    """

    def __init__(self):
        """
        初始化路由器

        注意：适配器的顺序很重要！
        - 先检查特定模式（A股、港股、商品）
        - 最后检查默认模式（美股）
        """
        self.adapters: List[MarketAdapter] = []
        self._initialize_adapters()

    def _initialize_adapters(self):
        """
        初始化所有市场适配器

        延迟导入避免循环依赖
        """
        # 延迟导入，避免循环依赖
        from src.markets.cn_stock import CNStockAdapter
        from src.markets.hk_stock import HKStockAdapter
        from src.markets.commodity import CommodityAdapter
        from src.markets.us_stock import USStockAdapter

        # 顺序很重要：先检查特定模式，最后检查默认
        self.adapters = [
            CNStockAdapter(),      # A股: 600000.SH, 000001.SZ
            HKStockAdapter(),      # 港股: 0700.HK
            CommodityAdapter(),    # 商品: GC=F
            USStockAdapter(),      # 美股: AAPL (默认)
        ]

    def route(self, ticker: str) -> MarketAdapter:
        """
        将ticker路由到对应的市场适配器

        Args:
            ticker: 股票/商品代码

        Returns:
            MarketAdapter: 支持该ticker的适配器实例

        Raises:
            ValueError: 如果没有适配器支持该ticker

        Examples:
            >>> router = MarketRouter()
            >>> adapter = router.route("600000.SH")
            >>> isinstance(adapter, CNStockAdapter)
            True
        """
        for adapter in self.adapters:
            if adapter.supports_ticker(ticker):
                return adapter

        raise ValueError(f"未找到支持该ticker的适配器: {ticker}")

    def get_prices(
        self,
        ticker: str,
        start_date: str,
        end_date: str
    ) -> List[Dict]:
        """
        便捷方法：获取价格数据

        自动路由到对应适配器并获取数据

        Args:
            ticker: 股票/商品代码
            start_date: 开始日期 "YYYY-MM-DD"
            end_date: 结束日期 "YYYY-MM-DD"

        Returns:
            List[Dict]: 价格数据列表
        """
        adapter = self.route(ticker)
        return adapter.get_prices(ticker, start_date, end_date)

    def get_company_news(
        self,
        ticker: str,
        end_date: str,
        limit: int
    ) -> List[Dict]:
        """
        便捷方法：获取新闻数据

        自动路由到对应适配器并获取数据

        Args:
            ticker: 股票/商品代码
            end_date: 截止日期 "YYYY-MM-DD"
            limit: 最大新闻数量

        Returns:
            List[Dict]: 新闻列表
        """
        adapter = self.route(ticker)
        return adapter.get_company_news(ticker, end_date, limit)

    def get_financial_metrics(
        self,
        ticker: str,
        end_date: str
    ) -> Dict:
        """
        便捷方法：获取财务指标

        自动路由到对应适配器并获取数据

        Args:
            ticker: 股票代码
            end_date: 截止日期 "YYYY-MM-DD"

        Returns:
            Dict: 财务指标字典
        """
        adapter = self.route(ticker)
        return adapter.get_financial_metrics(ticker, end_date)
```

```python
"""
A股市场适配器

使用AkShare获取A股（上海、深圳证券交易所）的市场数据
"""
import akshare as ak
from datetime import datetime
from typing import List, Dict
from src.markets.base import MarketAdapter


class CNStockAdapter(MarketAdapter):
    """
    A股市场适配器

    支持的ticker格式:
    - 上海证券交易所: 600000.SH, 600519.SH
    - 深圳证券交易所: 000001.SZ, 000002.SZ

    数据来源: AkShare (免费、开源)
    """

    def supports_ticker(self, ticker: str) -> bool:
        """
        检查是否支持该ticker

        支持格式:
        - .SH 后缀（上海证券交易所）
        - .SZ 后缀（深圳证券交易所）

        Args:
            ticker: 股票代码

        Returns:
            bool: True表示支持A股格式
        """
        return ticker.endswith('.SH') or ticker.endswith('.SZ')

    def get_prices(
        self,
        ticker: str,
        start_date: str,
        end_date: str
    ) -> List[Dict]:
        """
        获取A股历史价格数据

        使用AkShare的stock_zh_a_hist接口

        Args:
            ticker: 股票代码，如 "600000.SH"
            start_date: 开始日期 "YYYY-MM-DD"
            end_date: 结束日期 "YYYY-MM-DD"

        Returns:
            List[Dict]: 标准格式的价格数据

        Raises:
            Exception: 数据获取失败
        """
        # 从ticker中提取股票代码（去掉.SH或.SZ后缀）
        symbol = ticker.split('.')[0]

        # AkShare需要日期格式为YYYYMMDD
        start_str = start_date.replace('-', '')
        end_str = end_date.replace('-', '')

        try:
            # 调用AkShare接口获取数据
            df = ak.stock_zh_a_hist(
                symbol=symbol,
                start_date=start_str,
                end_date=end_str,
                adjust="qfq"  # 前复权，考虑拆股和分红
            )

            # 转换为标准格式
            return self._convert_prices_to_standard(df)

        except Exception as e:
            raise Exception(f"获取A股价格数据失败 {ticker}: {str(e)}")

    def get_company_news(
        self,
        ticker: str,
        end_date: str,
        limit: int
    ) -> List[Dict]:
        """
        获取A股公司新闻

        数据来源:
        1. 东方财富新闻 (主要来源)
        2. Google News RSS (备用来源)

        Args:
            ticker: 股票代码
            end_date: 截止日期
            limit: 最大新闻数量

        Returns:
            List[Dict]: 标准格式的新闻数据
        """
        symbol = ticker.split('.')[0]
        news_list = []

        try:
            # 主要来源：东方财富新闻
            df = ak.stock_news_em(symbol=symbol)
            news_list = self._convert_news_to_standard(df, limit)

        except Exception as e:
            # 备用来源：Google News RSS
            try:
                from src.tools.news_sources import GoogleNewsRSS
                rss = GoogleNewsRSS()
                news_list = rss.fetch_news(ticker, limit, language='zh-CN')
            except Exception as rss_error:
                # 两个来源都失败，返回空列表而不是抛出异常
                # 因为新闻不是必需的数据
                news_list = []

        return news_list

    def get_financial_metrics(
        self,
        ticker: str,
        end_date: str
    ) -> Dict:
        """
        获取A股财务指标

        使用AkShare的stock_financial_analysis_indicator接口

        Args:
            ticker: 股票代码
            end_date: 截止日期

        Returns:
            Dict: 财务指标字典
        """
        symbol = ticker.split('.')[0]

        try:
            # 获取财务分析指标
            df = ak.stock_financial_analysis_indicator(symbol=symbol)

            if df.empty:
                return {}

            # 获取最新的指标数据
            latest = df.iloc[-1]

            return {
                'pe_ratio': float(latest.get('pe_ratio', 0)),
                'pb_ratio': float(latest.get('pb_ratio', 0)),
                'market_cap': float(latest.get('market_cap', 0)),
                'revenue': float(latest.get('revenue', 0)),
                'net_profit': float(latest.get('net_profit', 0)),
            }

        except Exception as e:
            # 财务指标获取失败时返回空字典
            return {}

    def _convert_prices_to_standard(self, df) -> List[Dict]:
        """
        将AkShare的DataFrame转换为标准格式

        AkShare字段名为中文，需要映射到英文标准字段

        Args:
            df: AkShare返回的DataFrame

        Returns:
            List[Dict]: 标准格式的价格列表
        """
        result = []

        for _, row in df.iterrows():
            result.append({
                'date': row['日期'],
                'open': float(row['开盘']),
                'high': float(row['最高']),
                'low': float(row['最低']),
                'close': float(row['收盘']),
                'volume': int(row['成交量']),
            })

        return result

    def _convert_news_to_standard(self, df, limit: int) -> List[Dict]:
        """
        将AkShare的新闻DataFrame转换为标准格式

        Args:
            df: AkShare返回的新闻DataFrame
            limit: 最大新闻数量

        Returns:
            List[Dict]: 标准格式的新闻列表
        """
        result = []

        for _, row in df.head(limit).iterrows():
            result.append({
                'title': row['新闻标题'],
                'published': row['发布时间'],
                'source': row['新闻来源'],
                'sentiment': None,  # 情绪需要由LLM分析
            })

        return result
```

"""
港股市场适配器

使用Yahoo Finance (yfinance) 获取港股数据
"""
import yfinance as yf
from typing import List, Dict
from src.markets.base import MarketAdapter


class HKStockAdapter(MarketAdapter):
    """
    港股市场适配器

    支持的ticker格式:
    - 香港交易所: 0700.HK, 9988.HK, 0001.HK

    数据来源: Yahoo Finance (yfinance)
    """

    def supports_ticker(self, ticker: str) -> bool:
        """
        检查是否支持该ticker

        支持格式: .HK 后缀（香港交易所）

        Args:
            ticker: 股票代码

        Returns:
            bool: True表示支持港股格式
        """
        return ticker.endswith('.HK')

    def get_prices(
        self,
        ticker: str,
        start_date: str,
        end_date: str
    ) -> List[Dict]:
        """
        获取港股历史价格数据

        优先使用Yahoo Finance，如果失败则尝试AkShare港股接口

        Args:
            ticker: 股票代码，如 "0700.HK"
            start_date: 开始日期 "YYYY-MM-DD"
            end_date: 结束日期 "YYYY-MM-DD"

        Returns:
            List[Dict]: 标准格式的价格数据

        Raises:
            Exception: 所有数据源都失败时抛出异常
        """
        try:
            # 主要来源：Yahoo Finance
            df = yf.download(
                ticker,
                start=start_date,
                end=end_date,
                progress=False  # 禁用进度条
            )

            if df.empty:
                raise Exception("Yahoo Finance返回空数据")

            return self._convert_yf_to_standard(df)

        except Exception as yf_error:
            # 备用来源：AkShare港股接口
            try:
                import akshare as ak
                symbol = ticker.split('.')[0]
                df = ak.stock_hk_hist(
                    symbol=symbol,
                    start_date=start_date.replace('-', ''),
                    end_date=end_date.replace('-', '')
                )
                return self._convert_ak_to_standard(df)

            except Exception as ak_error:
                raise Exception(
                    f"获取港股价格数据失败 {ticker}: "
                    f"Yahoo Finance错误={str(yf_error)}, "
                    f"AkShare错误={str(ak_error)}"
                )

    def get_company_news(
        self,
        ticker: str,
        end_date: str,
        limit: int
    ) -> List[Dict]:
        """
        获取港股公司新闻

        使用Google News RSS（支持中英文）

        Args:
            ticker: 股票代码
            end_date: 截止日期
            limit: 最大新闻数量

        Returns:
            List[Dict]: 标准格式的新闻数据
        """
        try:
            from src.tools.news_sources import GoogleNewsRSS
            rss = GoogleNewsRSS()

            # 港股新闻使用中文搜索
            return rss.fetch_news(ticker, limit, language='zh-CN')

        except Exception as e:
            # 新闻获取失败返回空列表
            return []

    def get_financial_metrics(
        self,
        ticker: str,
        end_date: str
    ) -> Dict:
        """
        获取港股财务指标

        使用Yahoo Finance的info接口

        Args:
            ticker: 股票代码
            end_date: 截止日期

        Returns:
            Dict: 财务指标字典
        """
        try:
            stock = yf.Ticker(ticker)
            info = stock.info

            return {
                'pe_ratio': float(info.get('trailingPE', 0)),

"""
商品期货市场适配器

使用Yahoo Finance获取商品和期货数据
"""
import yfinance as yf
from typing import List, Dict
from src.markets.base import MarketAdapter


class CommodityAdapter(MarketAdapter):
    """
    商品期货市场适配器

    支持的ticker格式:
    - 期货: GC=F (黄金), SI=F (白银), CL=F (原油)
    - 现货: XAUUSD (黄金), XAGUSD (白银)

    数据来源: Yahoo Finance
    """

    # 支持的商品列表
    SUPPORTED_COMMODITIES = {
        'GC=F': '黄金期货',
        'SI=F': '白银期货',
        'CL=F': '原油期货',
        'XAUUSD': '黄金现货',
        'XAGUSD': '白银现货',
    }

    # 新闻搜索关键词映射
    COMMODITY_KEYWORDS = {
        'GC=F': 'gold price',
        'SI=F': 'silver price',
        'CL=F': 'crude oil price',
        'XAUUSD': 'gold',
        'XAGUSD': 'silver',
    }

    def supports_ticker(self, ticker: str) -> bool:
        """
        检查是否支持该ticker

        支持格式:
        - 期货: 包含 =F 后缀
        - 现货: XAUUSD, XAGUSD

        Args:
            ticker: 商品代码

        Returns:
            bool: True表示支持该商品格式
        """
        return '=F' in ticker or ticker in ['XAUUSD', 'XAGUSD']

    def get_prices(
        self,
        ticker: str,
        start_date: str,
        end_date: str
    ) -> List[Dict]:
        """
        获取商品期货历史价格数据

        使用Yahoo Finance

        Args:
            ticker: 商品代码，如 "GC=F"
            start_date: 开始日期 "YYYY-MM-DD"
            end_date: 结束日期 "YYYY-MM-DD"

        Returns:
            List[Dict]: 标准格式的价格数据

        Raises:
            Exception: 数据获取失败
        """
        try:
            df = yf.download(
                ticker,
                start=start_date,
                end=end_date,
                progress=False
            )

            if df.empty:
                raise Exception("Yahoo Finance返回空数据")

            return self._convert_to_standard(df)

        except Exception as e:
            raise Exception(f"获取商品价格数据失败 {ticker}: {str(e)}")

    def get_company_news(
        self,
        ticker: str,
        end_date: str,
        limit: int
    ) -> List[Dict]:
        """
        获取商品相关新闻

        使用Google News RSS，根据商品类型使用不同的搜索关键词

        Args:
            ticker: 商品代码
            end_date: 截止日期
            limit: 最大新闻数量

            return []

            end_date: 截止日期

            Dict: 空字典

        # 商品期货没有财务指标

                'volume': int(row['Volume']),
            })

        return result

```python
"""
新闻数据源

提供多种新闻获取方式，包括Google News RSS
"""
import feedparser
from typing import List, Dict
from datetime import datetime


class GoogleNewsRSS:
    """
    Google News RSS订阅源获取器

    优势:
    - 免费，无需API密钥
    - 无请求限制
    - 支持多种语言（中文、英文等）
    - 实时新闻更新

    使用示例:
        >>> rss = GoogleNewsRSS()
        >>> news = rss.fetch_news('AAPL', limit=10, language='en')
        >>> news = rss.fetch_news('600000.SH', limit=10, language='zh-CN')
    """

    BASE_URL = "https://news.google.com/rss/search"

    def fetch_news(
        self,
        query: str,
        limit: int = 20,
        language: str = 'zh-CN'
    ) -> List[Dict]:
        """
        从Google News RSS获取新闻

        Args:
            query: 搜索查询（ticker或关键词）
            limit: 最大新闻数量
            language: 语言代码
                - 'zh-CN': 简体中文
                - 'en': 英文
                - 'zh-TW': 繁体中文

        Returns:
            List[Dict]: 新闻列表，每个字典包含:
                - title: 新闻标题
                - published: 发布时间（ISO格式）
                - source: 新闻来源
                - link: 新闻链接
                - sentiment: 情绪（None，待LLM分析）

        Examples:
            >>> rss = GoogleNewsRSS()
            >>> news = rss.fetch_news('AAPL', limit=5, language='en')
            >>> len(news) <= 5
            True
        """
        # 构建搜索查询
        if language == 'zh-CN' or language == 'zh-TW':
            search_query = f"{query} 股票"
        else:
            search_query = f"{query} stock" if not query.endswith('stock') else query

        # 构建RSS URL参数
        params = {
            'q': search_query,
            'hl': language,
            'gl': 'CN' if language.startswith('zh') else 'US',
            'ceid': 'CN:zh-Hans' if language == 'zh-CN' else 'US:en'
        }

        # 生成完整URL
        url = f"{self.BASE_URL}?{'&'.join(f'{k}={v}' for k, v in params.items())}"

        try:
            # 解析RSS订阅
            feed = feedparser.parse(url)

            # 转换为标准格式
            news_list = []
            for entry in feed.entries[:limit]:
                news_list.append({
                    'title': entry.title,
                    'published': self._parse_date(entry.published),
                    'source': self._extract_source(entry),
                    'link': entry.link,
                    'sentiment': None,  # 情绪需要由LLM分析
                })

            return news_list

        except Exception as e:
            # RSS解析失败，返回空列表
            return []

    def _parse_date(self, date_str: str) -> str:
        """
        将RSS日期字符串解析为ISO格式

        Args:
            date_str: RSS日期字符串，如 "Mon, 01 Jan 2024 10:00:00 GMT"

        Returns:
            str: ISO格式日期字符串，如 "2024-01-01T10:00:00"
                 如果解析失败，返回原字符串

        Examples:
            >>> rss = GoogleNewsRSS()
            >>> rss._parse_date("Mon, 01 Jan 2024 10:00:00 GMT")
            '2024-01-01T10:00:00'
        """
        try:
            # 尝试解析RSS标准日期格式
            dt = datetime.strptime(date_str, '%a, %d %b %Y %H:%M:%S %Z')
            return dt.isoformat()
        except:
            # 解析失败，返回原字符串
            return date_str

    def _extract_source(self, entry) -> str:
        """
        从RSS条目中提取新闻来源

        Google News在标题中包含来源，格式为 "来源 - 文章标题"

        Args:
            entry: feedparser条目对象

        Returns:
            str: 新闻来源名称

        Examples:
            >>> rss = GoogleNewsRSS()
            >>> entry = Mock(title="Reuters - Breaking News")
            >>> rss._extract_source(entry)
            'Reuters'
        """
        title = entry.title

        # Google News标题格式: "来源 - 文章标题"
        if ' - ' in title:
            return title.split(' - ')[0]

        return 'Google News'
```

def get_prices(ticker: str, start_date: str, end_date: str, api_key: str = None):
    """
    获取历史价格数据（支持所有市场）

    此函数现在支持:
    - 美股（现有行为）
    - A股（600000.SH, 000001.SZ）
    - 港股（0700.HK）
    - 商品期货（GC=F）

    Args:
        ticker: 股票/商品代码
        start_date: 开始日期 "YYYY-MM-DD"
        end_date: 结束日期 "YYYY-MM-DD"
        api_key: API密钥（可选，仅美股Financial Datasets需要）

    Returns:
        List[dict]: 价格数据列表

    Examples:
        >>> # A股
        >>> get_prices("600000.SH", "2024-01-01", "2024-01-31")

        >>> # 港股
        >>> get_prices("0700.HK", "2024-01-01", "2024-01-31")

        >>> # 商品
        >>> get_prices("GC=F", "2024-01-01", "2024-01-31")

        >>> # 美股（需要api_key）
        >>> get_prices("AAPL", "2024-01-01", "2024-01-31", api_key="xxx")
    """
    try:
        # 首先尝试使用市场路由器（支持所有市场）
        return _market_router.get_prices(ticker, start_date, end_date)

    except Exception as router_error:
        # 如果路由器失败且提供了api_key，回退到Financial Datasets API
        if api_key:
            try:
                return _get_prices_from_financial_datasets(
                    ticker, start_date, end_date, api_key
                )
            except Exception as fd_error:
                raise Exception(
                    f"获取价格数据失败 {ticker}: "
                    f"Router错误={str(router_error)}, "
                    f"Financial Datasets错误={str(fd_error)}"
                )
        else:
            # 没有api_key，直接抛出router错误
            raise router_error
